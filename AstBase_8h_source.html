108template <class T> class LexicalStack : public std::stack<LexicalContext<T>> {
109public:
110 void push_context() {
111 if (this->empty())
112 this->push(LexicalContext<T>());
113 else
114 this->push(LexicalContext<T>(&this->top()));
115 }
116 void pop_context() {
117 if (this->empty())
118 sammine_util::abort("ICE: You are popping an empty lexical stack");
119 this->pop();
120 }
121
122 void registerNameT(const std::string &name, T l) {
123 return this->top().registerNameT(name, l);
124 }
125 NameQueryResult recursiveQueryName(const std::string &name) const {
126 return this->top().recursiveQueryName(name);
127 }
128
129 T get_from_name(const std::string &name) {
130 return this->top().get_from_name(name);
131 }
132 NameQueryResult queryName(const std::string &name) const {
133 return this->top().queryName(name);
134 }
135
136 T recursive_get_from_name(const std::string &name) {
137 return this->top().recursive_get_from_name(name);
138 }
139
140 LexicalContext<T> *parent_scope() { return this->top().parent_scope; }
